Simple, powerful API

Zero dependencies for the core. Optional React hooks and Zod schema binding.

import { StreamingJSONParser } from 'jsonsettle'

// Works with any LLM provider — OpenAI, Anthropic, Gemini, your own
const parser = new StreamingJSONParser({
  onUpdate({ partial, fields, isComplete }) {
    // Fires on every token
    for (const [key, info] of Object.entries(fields)) {
      if (info.certainty === 'certain') {
        // This field's value will NOT change anymore ✓
        renderField(key, info.value)
      }
    }
  }
})

// Feed any stream — ReadableStream, AsyncIterator, whatever
const response = await fetch('/api/llm')
const reader = response.body.getReader()
while (true) {
  const { done, value } = await reader.read()
  if (done) { parser.flush(); break }
  parser.write(new TextDecoder().decode(value))
}
import { useStreamingJSON } from 'jsonsettle/react'

function UserCard({ stream }) {
  const { partial, fields, isComplete } = useStreamingJSON<User>(stream)

  return (
    <div>
      <Field
        label="Name"
        value={partial.name}
        certainty={fields['name']?.certainty}
      />
      <Field
        label="Age"
        value={partial.age}
        certainty={fields['age']?.certainty}
      />
    </div>
  )
}
import { createSchemaParser } from 'jsonsettle/zod'
import { z } from 'zod'

const UserSchema = z.object({
  name:   z.string(),
  age:    z.number().int().positive(),
  email:  z.string().email(),
  active: z.boolean(),
})

const parser = createSchemaParser(UserSchema, {
  schema: UserSchema,
  // Fires only when a field is CERTAIN and passes schema validation
  onCertainField(key, value) {
    console.log(`✓ ${key}: ${value}`) // typed as z.infer<typeof UserSchema>
  },
})

Why not existing solutions?

✗ partial-json / @streamparser/json

  • Repairs broken JSON but has no concept of certainty
  • No schema awareness — doesn't know your types
  • No TypeScript inference on partial values
  • No field-level callbacks — you get the whole object or nothing
  • Can't tell you "this field will NOT change anymore"

✓ jsonsettle

  • Each field tagged streaming vs certain — never guess again
  • Schema-aware with Zod: validates certainty-gated fields against your types
  • Full TypeScript inference — Partial<YourType> out of the box
  • React hook that re-renders only on relevant field changes
  • One small runtime dependency (twine) — works in Node, Deno, Bun, browsers, edge
